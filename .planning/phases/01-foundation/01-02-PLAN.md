       type Mode int

       const (
           ModeConfirm Mode = iota // Default: reads auto-approve, writes confirm
           ModePlan                // Read-only: everything non-read denied

       )

           default: return ModeConfirm

    3. Create `internal/permission/engine.go`:
       ```go
       package permission

       import (
           "crypto/sha256"
           "encoding/json"
           "fmt"
       )

       type Decision int

       const (
           Allow   Decision = iota // Proceed without asking
           Confirm                 // Ask user for permission
           Deny                    // Block execution
       )

       // ToolRiskProvider is the subset of Tool interface needed by the engine.
       // Avoids circular dependency: permission does NOT import tools.
       type ToolRiskProvider interface {
           Name() string
           RiskLevel() RiskLevel
       }

       type Engine struct {
           mode  Mode
           cache map[string]Decision
       }

       func NewEngine(defaultMode Mode) *Engine {
           return &Engine{mode: defaultMode, cache: make(map[string]Decision)}
       }

       func (e *Engine) Check(tool ToolRiskProvider, input json.RawMessage) Decision {
           risk := tool.RiskLevel()
           switch e.mode {
           case ModePlan:
               if risk > RiskReadOnly {
                   return Deny
               }
               return Allow
           case ModeBypass:
               return Allow
           case ModeConfirm:
               if risk <= RiskReadOnly {
                   return Allow
               }
               key := cacheKey(tool.Name(), input)
               if d, ok := e.cache[key]; ok {
                   return d
               }
               return Confirm
           }
           return Deny
       }

       func (e *Engine) CacheDecision(toolName string, input json.RawMessage, decision Decision) {
           key := cacheKey(toolName, input)
           e.cache[key] = decision
       }

       func (e *Engine) SetMode(mode Mode) { e.mode = mode }
       func (e *Engine) Mode() Mode         { return e.mode }
       func (e *Engine) CycleMode()         { e.mode = CycleMode(e.mode) }

       func cacheKey(name string, input json.RawMessage) string {
           h := sha256.Sum256(append([]byte(name+":"), input...))
           return fmt.Sprintf("%x", h[:8])
       }
       ```
       NOTE: Use `ToolRiskProvider` interface (not importing tools.Tool) to avoid circular dependency between permission and tools packages.
